C: Sorting Eigen Vectors by Eigen Values descending

eigVectOrg = eigVect;
d = diag(eigVals);
[~, ind] = sort(d, 'descend');
eigVect = eigVect(:,ind)

eigVect =

   -0.1376    0.6990    0.7017
   -0.2505    0.6609   -0.7075
